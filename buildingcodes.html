Structures Simples

Affichage - Partie 1

Affichage d'un texte simple

Afficher("J'aime la vie!")
print("J'aime la vie!")

Affichage du contenu d'une variable

msg ← "J'aime la vie!"
Afficher(msg)
msg = "J'aime la vie"
print(msg)

Affichage mixte (textes et variables)

nom ← "Mohamed"
Afficher(nom, "est plein d'espoir.")
nom = "Mohamed"
print(nom, "est plein d'espoir.")
nom ← "Mohamed"
age ← 16
Afficher("Je suis", nom, ", j'ai", age, "ans.")
nom = "Mohamed"
age = 16
print("Je suis", nom, ", j'ai", age, "ans.")

Affichage - Partie 2

f-String simple

Afficher une expression

x ← 8
y ← 5
Afficher(x, "*", y, "=", x*y)
x = 8
y = 5
print(f'{x} * {y} = {x*y}')  # 8 * 5 = 40

f-String entiers

Dans {x:5} le nombre 5 indique la taille minimale du champ x.

x ← 1
y ← 2
Afficher("x=", x, "- y=", y)
x = 1
y = 2
print(f'x= {x:5} - y= {y:5}')  # x=     1 - y=     2

Dans {h:02}, h est le nom d'une variable, 0 est le caractère de remplissage et 2 la taille minimale à réserver pour l'affichage de h.

h ← 7
m ← 0
s ← 3
Afficher(h, ":", m, ":", s)
h = 7
m = 0
s = 3
print(f'{h:02}:{m:02}:{s:02}')  # 07:00:03

f-String réels

Dans {moy:.2f} le .2 indique le nombre de chiffres après la virgule à utiliser pour l'affichage d'un réel.

dc ← 11
ds ← 12
moy ← (dc + ds * 2) / 3
Afficher("Moyenne =", moy)
dc = 11
ds = 12
moy = (dc + ds * 2) / 3
print(f'Moyenne = {moy:.2f}')  # Moyenne = 11.67

Structures conditionnelles

Si

Forme réduite

// Valeur absolue d'un nombre
a ← -9
va ← a
si a < 0 alors
    a ← -a
fin si
Afficher("|", a, "|=", va)
# Valeur absolue d'un nombre
a = -9
va = a
if a < 0:
    va = -a

print(f'| {a} |= {va}')  # | -9 |= 9

Forme alternative

// Signe d'un nombre

a ← alea(-10, 10)
si a >= 0 alors
    afficher(a, "positif")
sinon
    afficher(a, "négatif")
fin si
# Signe d'un nombre
from random import randint
a = randint(-10, 10)
if a >= 0:
    print(f'{a} positif')
else:
    print(f'{a} négatif')
# -3 négatif

Forme complète

// Comparaison de deux nombres

a ← alea(1, 5)
b ← alea(1, 5)
si a > b alors
    afficher(a, ">", b)
sinon si a < b alors
    afficher(a, "<", b)
sinon
    afficher(a, "=", b)
fin si
# Comparaison de deux nombres
from random import randint
a = randint(1, 5)
b = randint(1, 5)
if a > b:
    print(a, '>', b)
elif a < b:
    print(a, '<', b)
else:
    print(a, '=', b)
# 4 > 1

Structures itératives

Pour

Afficher un message

// Afficher un message plusieurs fois
Pour i de 0 à 4 Faire
    Afficher("I am a Pythonista!")
Fin Pour
# Afficher un message plusieurs fois
for i in range(5):
    print("I am a Pythonista!")

Fin Pour

for i in range(10, 21):
    print(f"Le compteur marque {i}")

Les nombres d'un intervalle par pas fixe

Fin Pour

for i in range(3, 21, 4):
    print(f"Le compteur marque {i}")
