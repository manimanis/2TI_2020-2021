Affichage - Partie 2

f-String simple

Afficher une expression

x ← 8
y ← 5
Ecrire(x, "*", y, "=", x*y)
x = 8
y = 5
print(f'{x} * {y} = {x*y}')  # 8 * 5 = 40

f-String entiers

Dans {x:5} le nombre 5 indique la taille minimale du champ x.

x ← 1
y ← 2
Ecrire("x=", x, "- y=", y)
x = 1
y = 2
print(f'x= {x:5} - y= {y:5}')  # x=     1 - y=     2

Dans {h:02}, h est le nom d'une variable, 0 est le caractère de remplissage et 2 la taille minimale à réserver pour l'affichage de h.

h ← 7
m ← 0
s ← 3
Ecrire(h, ":", m, ":", s)
h = 7
m = 0
s = 3
print(f'{h:02}:{m:02}:{s:02}')  # 07:00:03

f-String réels

Dans {moy:.2f} le .2 indique le nombre de chiffres après la virgule à utiliser pour l'affichage d'un réel.

dc ← 11
ds ← 12
moy ← (dc + ds * 2) / 3
Ecrire("Moyenne =", moy)
dc = 11
ds = 12
moy = (dc + ds * 2) / 3
print(f'Moyenne = {moy:.2f}')  # Moyenne = 11.67

Structures conditionnelles

Si

Forme réduite

si [condition] alors
    // Traitements
fin si
if [condition]:
    # Traitements

Exemple

// Valeur absolue d'un nombre
a ← -9
va ← a
si a < 0 alors
    a ← -a
fin si
Ecrire("|", a, "|=", va)
# Valeur absolue d'un nombre
a = -9
va = a
if a < 0:
    va = -a

print(f'| {a} |= {va}')  # | -9 |= 9

Forme alternative

si [condition] alors
    // Traitement, si [condition] est vraie
sinon
    // Traitement, si [condition] est fausse
fin si
if [condition]:
    # Traitements, si [condition] est vraie
else:
    # Traitements, si [condition] est fausse

Exemple

// Signe d'un nombre

a ← alea(-10, 10)
si a >= 0 alors
    Ecrire(a, "positif")
sinon
    Ecrire(a, "négatif")
fin si
# Signe d'un nombre
from random import randint
a = randint(-10, 10)
if a >= 0:
    print(f'{a} positif')
else:
    print(f'{a} négatif')
# -3 négatif

Forme complète

si [condition1] alors
    // Traitement, si [condition1] est vraie
sinon si [condition2] alors
    // Traitement, si [condition2] est vraie
sinon
    // Traitement, si toutes les condition sont fausses
fin si
if [condition1]:
    # Traitements, si [condition1] est vraie
elif [condition2]:
    # Traitements, si [condition2] est vraie
else:
    # Traitements, si toutes les condition sont fausses

Exemple

// Comparaison de deux nombres

a ← alea(1, 5)
b ← alea(1, 5)
si a > b alors
    Ecrire(a, ">", b)
sinon si a < b alors
    Ecrire(a, "<", b)
sinon
    Ecrire(a, "=", b)
fin si
# Comparaison de deux nombres
from random import randint
a = randint(1, 5)
b = randint(1, 5)
if a > b:
    print(a, '>', b)
elif a < b:
    print(a, '<', b)
else:
    print(a, '=', b)
# 4 > 1

Structures itératives

Pour

1ère forme

pour [cpt] de 0 à [fin]-1 Faire
    // Traitements
fin pour
for [cpt] in range([fin]):
    # Traitements

Exemple : Afficher un message

// Afficher un message plusieurs fois
Pour i de 0 à 4 Faire
    Ecrire("I am a Pythonista!")
Fin Pour
# Afficher un message plusieurs fois
for i in range(5):
    print("I am a Pythonista!")

2ème forme

pour [cpt] de [debut] à [fin] Faire
    // Traitements
fin pour
for [cpt] in range([debut], [fin]+1):
    # Traitements

Exemple : Les nombres d'un intervalle

// Afficher tous les nombres d'un intervalle
Pour i de 10 à 20 Faire
    Ecrire("Le compteur marque", i)
Fin Pour
# Afficher un message plusieurs fois
for i in range(10, 21):
    print(f"Le compteur marque {i}")

3ème forme

pour [cpt] de [debut] à [fin] (pas=[p]) Faire
    // Traitements
fin pour
for [cpt] in range([debut], [fin]+1, [p]):
    # Traitements

Exemple : Les nombres d'un intervalle par pas fixe 

// Les nombres d'un intervalle par pas de 4
Pour i de 3 à 20 (pas = 4) Faire
    Ecrire("Le compteur marque", i)
Fin Pour
# Afficher un message plusieurs fois
for i in range(3, 21, 4):
    print(f"Le compteur marque {i}")

Répéter ... Jusqu'à

Répéter
    // Traitements
Jusqu'à [condition]
while True:
    # Traitements
    if [condition]:
        break

Exemple : Saisir un entier de deux chiffres

Répéter
    Ecrire("Donner un entier [10, 99] ? ")
    Lire(n)
Jusqu'à 10 ≤ n ≤ 99
 
while True:
    n = int(input("Donner un entier [10, 99] ? "))
    
    if 10 <= n <= 99:
        break

Tant que

Tant que [condition] Faire
    // Traitements
Fin Tant que
while [condition]:
    # Traitements

Exemple : Calculer le PGCD de deux nombres


a ← alea(10, 50)
b ← alea(10, 50)
Ecrire("a=", a, "- b=", b)
Tant que b ≠ 0 Faire
    r ← a mod b
    a ← b
    b ← r
Fin Tant Que
// Le PGCD est dans la variable a
Ecrire(a)
from random import randint
a = randint(10, 50)
b = randint(10, 50)
print("a=", a, "- b=", b)
while b != 0:
    r = a % b
    a = b
    b = r

# Le PGCD est dans la variable a
print(a)
